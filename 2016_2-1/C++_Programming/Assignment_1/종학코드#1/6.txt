#include <iostream>
using namespace std;

int main()
{
	int num, cnt = 0;
	cout << "Input the size of array : ";
	cin >> num;
	if (num < 3)
	{
		cout << "It can't make a triagle" << endl;
		return 0;
	}
	int * arr = new int[num];//메모리 동적할당
	cout << "Input : ";
	for (int i = 0; i < num; i++)
	{
		cin >> arr[i];
	}
	for (int i = 0; i < num - 1; i++)//버블소트
	{
		for (int j = 0; j < num - i -1; j++)
		{
			if (arr[j] < arr[j + 1])
			{
				int temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;
			}
		}
	}
	for (int i = 0; i < num - 2; i++)//삼각형검사기
	{
		for (int j = i + 1; j < num - 1; j++)
		{
			for (int k = j + 1; k < num; k++)
			{
				if (arr[i] < arr[j] + arr[k])//세 변의 길이를 이용
					cnt++;
			}
		}
	}
	cout << "Output : " << cnt << endl;
	delete[] arr;//메모리 할당 해제
	return 0;

}
