#include <iostream>
using namespace std;

int main()
{
	char str[100];
	int count[26] = { 0, };
	cout << "Input : ";
	cin.getline(str,100);//엔터 입력시까지 입력
	int num = strlen(str);
	if (num > 99)//에러
	{
		cout << "Wrong Input" << endl;
		return 0;
	}
	for (int i = 0; i < strlen(str); i++)
	{
		if (str[i] == ' ')//띄어쓰기 검사
			num--;
		else
			count[str[i] - 'a'] += 1;//문자 검사
	}
	for (int i = 0; i < 26; i++)//출력
	{
		if (count[i]!=0)
			cout << (char)(i + 'a') << '\t' << (float)count[i] / (float)num << endl;
	}
	return 0;
}
