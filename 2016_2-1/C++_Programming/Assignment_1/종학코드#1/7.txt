#include <iostream>
using namespace std;

int main()
{
	int num, sum = 0, pcheck = 0;
	cout << "The number is ";
	cin >> num;
	if (num < 1 || num>9999)//입력 범위 밖의 입력시 종료
	{
		cout << "Wrong number" << endl;
		return 0;
	}
	for (int i = 0; i < 3; i++)//a검사(나누기 검사)
	{
		int seq[3] = { 7, 11, 13 };
		if (num % seq[i] == 0)
			cout << "It can be divided by " << seq[i] << ", the result is " << num / seq[i] << endl;
		else
			cout << "It cannot be divided by " << seq[i] << endl;
	}
	int tnum = num;//num을 변화시키지 않기 위해 임시 저장값
	while (tnum)//digit의 합
	{
		sum += tnum % 10;
		tnum /= 10;
	}
	char *what;
	if (sum % 2 == 0)//digit의 합 검사후 짝홀 검사
		what = "even";
	else
		what = "odd";
	cout << "The sum of every digit is " << sum << ", it is " << what << " number" << endl;
	for (int i = 2; i < num / 2 + 1; i++)//소수여부검사
	{
		if (num%i == 0)
			pcheck++;
	}
	if (pcheck != 0)
		cout << "It is not a prime number" << endl;
	else
		cout << "It is a prime number" << endl;
	return 0;
}
