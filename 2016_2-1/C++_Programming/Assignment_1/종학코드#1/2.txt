#include <iostream>
using namespace std;

int main()
{
	int point[3][2], i, j, sqr[2];
	cout << "Input" << endl;
	for (i = 0; i < 3; i++)//입력부
	{
		cout << "[" << i+1 << "] : ";
		for (j = 0; j < 2; j++)
		{
			cin >> point[i][j];
		}
	}
	for (int i = 0, j=1; i < 3; i++, j++)//주어진 3점중에 똑같은 점이 있을시 에러처리
	{
		if (j == 3)
			j = 0;
		if (point[i][0] == point[j][0] && point[i][1] == point[j][1])
		{
			cout << "Wrong Input" << endl;
			return 0;
		}
	}
	for (int k = 0; k < 2; k++)//반복은 각각 x축,y축 검사
	{
		i = 0;	j = 1;
		while (1)
		{
			if (point[i][k] == point[j][k])//축과 평행한지 검사
			{
				sqr[k] = point[3 - i - j][k];
				break;
			}
			i++; j++;
			if (j == 3)
				j = 0;
			if (i == 3)//error
			{
				cout << "Output" << endl << "Not rectangle" << endl;
				return 0;
			}
		}
	}
	cout << "Output" << endl << "[4] : " << sqr[0] << " " << sqr[1] << endl;//출력
	return 0;
}
