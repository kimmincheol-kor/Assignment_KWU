#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	ofstream fout;
	fout.open("Calendar.txt");//파일생성

	int year, month, sur, date;
	int mdate[12] = {  31,28,31,30,31,30,31,31,30,31,30,31 };//매 달마다의 일반적인 날의 수
	cout << "Enter the year : ";
	cin >> year;
	if (year < 2010 || year>2020)//년도 범위 밖의 입력
	{
		cout << "Wrong number" << endl;
		return 0;
	}
	cout << "Enter the month : ";
	cin >> month;
	if (month < 1 || month>12)//달의 범위 밖의 입력
	{
		cout << "Wrong number" << endl;
		return 0;
	}
	sur = (year - 2008) / 4;//윤달
	if ((year - 2008) % 4 == 0 && month < 3)//2월이 지나기 전에는 윤달을 지난날에 포함시킬수 없으므로
		sur--;
	date = (year - 2010) * 365 + sur; //시작 기준날부터 지난 해의 날
	for (int i = 0; i < month - 1; i++)
		date += mdate[i];//지난 달의 날
	date += 4;//시작 기준날은 금요일이므로
	date %= 7;//일주일은 7일 이므로
	fout << year << "년" << month << "월" << endl;
	for (int i = 0; i < date; i++)
		fout << " " << '\t';//1일 전의 빈공간
	for (int i = 0; i < mdate[month - 1]; i++)
	{//월화수목금토일
		fout << i + 1 << '\t';
		date++;
		if (date % 7 == 0)
			fout << endl;
	}

	fout.close();
	return 0;
}
