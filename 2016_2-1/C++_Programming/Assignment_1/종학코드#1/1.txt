#include <iostream>
using namespace std;

int main()
{
	char word[100];
	cout << "Input : ";
	cin >> word;
	int num = strlen(word);//입력된 글자 수
	if (num > 99)//너무 많은 글자를 입력할 시
	{
		cout << "Wrong Input" << endl;
		return 0;
	}
	for (int i = 0; i < num; i++)
	{
		for (int j = 0; j < 3; j++)
			cout << word[(i + j) % num];
		cout << endl;
	}
	return 0;
}
